{Winning Strategy 3x3 Red}
[Win if possible]
P(0, x) -> -P(0, x)
P(1, x) AND E(0, x) -> -P(1, x)
E(0, x) AND E(1, x) AND P(2, x) -> -P(2, x)
[Attack]
E(0, x) AND P(1, x) -> +P(0, x) AND -P(1, x)
[Move]
P(1, 2) ∧ !E(0, 1) ∧ !P(0, 1) -> +P(0, 1) ∧ -P(1, 2)
P(2, 1) ∧ !E(1, 2) ∧ !P(1, 2) ∧ !E(0, 2) -> +P(1, 2) ∧ -P(2, 1)
[Place piece]
!PE(2, 1) ∧ !E(1, 1) -> +P(2, 1)
!PE(2, 0) ∧ !E(1, 0) -> +P(2, 0)
